Demonstrates the usage of a queue. This function creates a queue, adds elements to it, and then prints the elements.

#include <iostream>
#include <queue>
using std::queue;
using std::cout;
void printQueue(queue<int> queue) {
while (!queue.empty()) {
cout << queue.front() << "\n";
queue.pop();
}
}
